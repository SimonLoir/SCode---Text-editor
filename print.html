def fact(n):
    
    i = n
    
    p = 1  
    
    while i > 0:
        p = p * i
        i = i - 1
        
    return p #p est un entier qui pour tout i a été multiplié par i. i est un entier appartenant à R+ auquel on a retiré n * 1
         
if __name__ == "__main__":
    main()
